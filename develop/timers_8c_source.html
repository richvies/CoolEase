44// Start Low Speed Oscillator and Configure RTC to wakeup device
45void timers_rtc_init(void) {
46 // Check if rtc already initialized
47 if (RTC_ISR & RTC_ISR_INITS) {
48 serial_printf("RTC: Init\n");
49 } else {
50 log_printf("RTC: Not init\n");
51
52 // Enable PWR clock and disable write protection
53 // RCC_CSR Write protected. Must come before turning on LSI/ LSE clocks
54 timers_rtc_unlock();
55
56 // Start low speed internal oscillator ≈ 40kHz. And wait until it is
57 // ready
58 rcc_osc_on(RCC_LSI);
59 rcc_wait_for_osc_ready(RCC_LSI);
60
61 // Get LSI frequency
62 uint32_t lsi_freq = timers_measure_lsi_freq();
63 // serial_printf("Freq: %u\n", lsi_freq);
64
65 // Select LSI as RTC clk
66 RCC_CSR &= ~(RCC_CSR_RTCSEL_MASK << RCC_CSR_RTCSEL_SHIFT);
67 RCC_CSR |= (RCC_CSR_RTCSEL_LSI << RCC_CSR_RTCSEL_SHIFT);
68
69 // Enable RTC
70 RCC_CSR |= RCC_CSR_RTCEN;
71
72 timers_delay_milliseconds(10);
73
74 // Set RTC initialization mode
75 RTC_ISR |= RTC_ISR_INIT;
76 while (!((RTC_ISR) & (RTC_ISR_INITF))) {
77 }
78
79 // Set RTC prescaler
80 rtc_set_prescaler(((lsi_freq / 4) - 1), 0x0003);
81 // PRINT_REG(RTC_PRER);
82
83 // Exit RTC initialization mode
84 RTC_ISR &= ~RTC_ISR_INIT;
85
86 // Reenable write protection
87 timers_rtc_lock();
88
89 timers_rtc_set_time(21, 6, 24, 16, 20, 00);
90
91 log_printf(".done\n");
92 }
93}

95void timers_rtc_set_time(uint8_t year, uint8_t month, uint8_t day,
96 uint8_t hours, uint8_t mins, uint8_t secs) {
97 // Enable PWR clock and disable write protection
98 timers_rtc_unlock();
99
100 // Set shadow registers
101 uint32_t tr = 0;
102 uint32_t dr = 0;
103 tr |= (((secs % 10) & RTC_TR_SU_MASK) << RTC_TR_SU_SHIFT);
104 tr |= (((secs / 10) & RTC_TR_ST_MASK) << RTC_TR_ST_SHIFT);
105 tr |= (((mins % 10) & RTC_TR_MNU_MASK) << RTC_TR_MNU_SHIFT);
106 tr |= (((mins / 10) & RTC_TR_MNT_MASK) << RTC_TR_MNT_SHIFT);
107 tr |= (((hours % 10) & RTC_TR_HU_MASK) << RTC_TR_HU_SHIFT);
108 tr |= (((hours / 10) & RTC_TR_HT_MASK) << RTC_TR_HT_SHIFT);
109
110 dr |= (((day % 10) & RTC_DR_DU_MASK) << RTC_DR_DU_SHIFT);
111 dr |= (((day / 10) & RTC_DR_DT_MASK) << RTC_DR_DT_SHIFT);
112 dr |= (((month % 10) & RTC_DR_MU_MASK) << RTC_DR_MU_SHIFT);
113 dr |= (((month / 10) & RTC_DR_MT_MASK) << RTC_DR_MT_SHIFT);
114 dr |= (((year % 10) & RTC_DR_YU_MASK) << RTC_DR_YU_SHIFT);
115 dr |= (((year / 10) & RTC_DR_YT_MASK) << RTC_DR_YT_SHIFT);
116
117 // Set RTC initialization mode
118 RTC_ISR |= RTC_ISR_INIT;
119 while (!((RTC_ISR) & (RTC_ISR_INITF))) {
120 }
121
122 RTC_TR = tr;
123 RTC_DR = dr;
124
125 // Exit RTC initialization mode
126 RTC_ISR &= ~RTC_ISR_INIT;
127
128 // PRINT_REG(RTC_TR);
129 // PRINT_REG(RTC_DR);

132 timers_rtc_lock();

190static uint32_t timers_measure_lsi_freq(void) {
191 // TIM21 on APB2
192 rcc_periph_clock_enable(RCC_TIM21);
193 rcc_periph_reset_pulse(RST_TIM21);
194 timer_disable_counter(TIM21);
195
196 // Edge aligned, up counter
197 TIM_CR1(TIM21) |= TIM_CR1_CMS_EDGE | TIM_CR1_DIR_UP;
198
199 // Enable
200 TIM_CR1(TIM21) |= TIM_CR1_CEN;
201
202 // CC1 input capture mode, 8 prescale, 8 sample filter
203 TIM_CCMR1(TIM21) |=
204 TIM_CCMR1_CC1S_IN_TI1 | TIM_CCMR1_IC1PSC_8 | TIM_CCMR1_IC1F_CK_INT_N_8;
205
206 // Select LSI capture
207 TIM21_OR |= TIM21_OR_TI1_RMP_LSI;
208
209 // Enable capture
210 TIM_CCER(TIM21) |= TIM_CCER_CC1E;
211
212 // Wait for captures
213 uint32_t count = 0;
214 while (!(TIM_SR(TIM21) & TIM_SR_CC1IF)) {
215 };
216 uint16_t time = TIM_CCR1(TIM21);
217 for (uint16_t i = 0; i < 10; i++) {
218 while (!(TIM_SR(TIM21) & TIM_SR_CC1IF)) {
219 };
220 count += (uint16_t)(TIM_CCR1(TIM21) - time);
221 time = TIM_CCR1(TIM21);
222 }
223
224 // prescaler * number of samples
225 count /= 80;
226
227 uint32_t freq = rcc_apb2_frequency / count;
228
229 // Disable TIM21
230 TIM_CCER(TIM21) &= ~TIM_CCER_CC1E;
231 TIM_CR1(TIM21) &= ~TIM_CR1_CEN;
232 rcc_periph_reset_pulse(RST_TIM21);
233 rcc_periph_clock_disable(RCC_TIM21);
234
235 return freq;
236}

238void timers_lptim_init(void) {
239 // Internal clock = APB1
240 rcc_set_peripheral_clk_sel(LPTIM1, RCC_CCIPR_LPTIM1SEL_APB);
241
242 rcc_periph_clock_enable(RCC_LPTIM1);
243
244 // Reset timer
245 rcc_periph_reset_pulse(RST_LPTIM1);
246
247 // Select internal clock as src
248 lptimer_set_internal_clock_source(LPTIM1);
249
250 // Find best prescaler for 1us clock
251 uint8_t freq_mhz = rcc_apb1_frequency / 1000000;
252 for (uint8_t i = 0; i < 8; i++) {
253 if (freq_mhz <= (1 << i)) {
254 lptimer_set_prescaler(LPTIM1, (i << LPTIM_CFGR_PRESC_SHIFT));
255 break;
256 }
257 }
258
259 // Enable SW start triggering
260 lptimer_enable_trigger(LPTIM1, LPTIM_CFGR_TRIGEN_SW);
261
262 lptimer_enable(LPTIM1);
263
264 // Must be done after timer enable
265 lptimer_set_period(LPTIM1, 1000);
266 lptimer_enable_irq(LPTIM1, LPTIM_IER_ARRMIE);
267
268 micros_counter = 0;
269 millis_counter = 0;
270
271 // Enable lptim interrupt through exti29 - updates millis counter
272 nvic_enable_irq(NVIC_LPTIM1_IRQ);
273 nvic_set_priority(NVIC_LPTIM1_IRQ, IRQ_PRIORITY_LPTIM);
274 exti_reset_request(EXTI29);
275 exti_enable_request(EXTI29);
276 exti_set_trigger(EXTI29, EXTI_TRIGGER_RISING);
277
278 // Trigger SW start
279 lptimer_start_counter(LPTIM1, LPTIM_CR_CNTSTRT);
280}
