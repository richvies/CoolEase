157bool boot_program_half_page(bool lower, uint32_t crc_expected,
158 uint32_t page_num, uint32_t data[16]) {

173 log_error(ERR_USB_PROGRAM_UPPER_HALF_PAGE_FAIL);
174 }
175 }
176 // Everything went well
177 else {
178 success = true;
179 }

184bool boot_program_application(uint32_t* data, uint32_t start_address,
185 uint32_t len, uint32_t crc) {
186 log_printf("boot_program_application");
187
188 bool res = true;
189
195 if (start_address < FLASH_APP_ADDRESS || start_address > FLASH_END) {
196 log_error(ERR_BOOT_PROG_START_ADDRESS_OUT_OF_BOUNDS);
197 res = false;
198 } else if (len > FLASH_APP_ADDRESS - FLASH_APP_END) {
199 log_error(ERR_BOOT_PROG_TOO_BIG);
200 res = false;
201 } else if (!boot_verify_checksum(data, len, crc)) {
202 log_error(ERR_BOOT_PROG_BAD_CHECKSUM);
203 res = false;
204 }
205
207 else {
208 // Num pages to program
209 uint16_t num_pages = ((len / FLASH_PAGE_SIZE) + 1);
210 log_printf("Prog %i pgs\n", num_pages);
211
212 for (uint32_t address = start_address; address < start_address + len;
213 address += FLASH_PAGE_SIZE) {
214 // Erase page first
215 if (!mem_flash_erase_page(address)) {
216 log_error(ERR_BOOT_PROG_FLASH_ERASE);
217 res = false;
218 }
219 // Flash first half page
220 else if (!mem_flash_write_half_page(address, &data[0])) {
221 log_error(ERR_BOOT_PROG_FLASH_WRITE_1);
222 res = false;
223 }
224 // Flash second half page
225 else if (!mem_flash_write_half_page(address, &data[16])) {
226 log_error(ERR_BOOT_PROG_FLASH_WRITE_2);
227 res = false;
228 }
229
230 // Stop programming if flash error
231 if (res == false) {
232 break;
233 }
234 }
235 }
236
237 return res;
238}

240bool boot_verify_checksum(uint32_t* data, uint32_t len, uint32_t expected) {
241 log_printf("boot_verify_checksum\n");
242
243 // Initialize CRC Peripheral
244 rcc_periph_clock_enable(RCC_CRC);
245 crc_reset();
246 crc_set_reverse_input(CRC_CR_REV_IN_BYTE);
247 crc_reverse_output_enable();
248
249 // Calc CRC32
250 uint32_t crc = ~crc_calculate_block(data, len);
251
252 // Deinit
253 crc_reset();
254 rcc_periph_clock_disable(RCC_CRC);
255
256 serial_printf("Checksum value: %8x\n", crc);
257
258 // Check against expected
259 return (crc == expected ? true : false);
260}
