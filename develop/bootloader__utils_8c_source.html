179bool boot_program_half_page(bool lower, uint32_t crc_expected,
180 uint32_t page_num, uint32_t data[16]) {
181 bool success = false;
182
183 // Check crc32
184 if (!verify_half_page_checksum(data, crc_expected)) {
185 log_error(ERR_USB_PAGE_CHECKSUM_BAD);
186 }
187 // Program half page
188 else if (!mem_flash_write_half_page(FLASH_APP_ADDRESS +
189 (page_num * FLASH_PAGE_SIZE) +
190 (lower ? 0 : (FLASH_PAGE_SIZE / 2)),
191 data)) {
192 if (lower) {
193 log_error(ERR_USB_PROGRAM_LOWER_HALF_PAGE_FAIL);
194 } else {
195 log_error(ERR_USB_PROGRAM_UPPER_HALF_PAGE_FAIL);
196 }
197 }
198 // Everything went well
199 else {
200 success = true;
201 }
202
203 return success;
204}

206bool boot_program_application(uint32_t* data, uint32_t start_address,
207 uint32_t len, uint32_t crc) {
208 log_printf("boot_program_application");
209
210 bool res = true;
211
217 if (start_address < FLASH_APP_ADDRESS || start_address > FLASH_END) {
218 log_error(ERR_BOOT_PROG_START_ADDRESS_OUT_OF_BOUNDS);
219 res = false;
220 } else if (len > FLASH_APP_ADDRESS - FLASH_APP_END) {
221 log_error(ERR_BOOT_PROG_TOO_BIG);
222 res = false;
223 } else if (!boot_verify_checksum(data, len, crc)) {
224 log_error(ERR_BOOT_PROG_BAD_CHECKSUM);
225 res = false;
226 }
227
229 else {
230 // Num pages to program
231 uint16_t num_pages = ((len / FLASH_PAGE_SIZE) + 1);
232 log_printf("Prog %i pgs\n", num_pages);
233
234 for (uint32_t address = start_address; address < start_address + len;
235 address += FLASH_PAGE_SIZE) {
236 // Erase page first
237 if (!mem_flash_erase_page(address)) {
238 log_error(ERR_BOOT_PROG_FLASH_ERASE);
239 res = false;
240 }
241 // Flash first half page
242 else if (!mem_flash_write_half_page(address, &data[0])) {
243 log_error(ERR_BOOT_PROG_FLASH_WRITE_1);
244 res = false;
245 }
246 // Flash second half page
247 else if (!mem_flash_write_half_page(address, &data[16])) {
248 log_error(ERR_BOOT_PROG_FLASH_WRITE_2);
249 res = false;
250 }
251
252 // Stop programming if flash error
253 if (res == false) {
254 break;
255 }
256 }
257 }
258
259 return res;
260}

262bool boot_verify_checksum(uint32_t* data, uint32_t len, uint32_t expected) {
263 log_printf("boot_verify_checksum\n");
264
265 // Initialize CRC Peripheral
266 rcc_periph_clock_enable(RCC_CRC);
267 crc_reset();
268 crc_set_reverse_input(CRC_CR_REV_IN_BYTE);
269 crc_reverse_output_enable();
270
271 // Calc CRC32
272 uint32_t crc = ~crc_calculate_block(data, len);
273
274 // Deinit
275 crc_reset();
276 rcc_periph_clock_disable(RCC_CRC);
277
278 serial_printf("Checksum value: %8x\n", crc);
279
280 // Check against expected
281 return (crc == expected ? true : false);
282}
