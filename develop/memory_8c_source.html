98bool mem_eeprom_write_word(uint32_t address, uint32_t data) {
99 if (!(address >= EEPROM_START && address < EEPROM_END))
100 return false;
101
102 if (MMIO32(address) == data)
103 return true;
104
105 flash_unlock_pecr();
106 FLASH_PECR &= ~FLASH_PECR_FTDW;
107
108 MMIO32(address) = data;
109
110 while (FLASH_SR & FLASH_SR_BSY)
111 ;
112 if (FLASH_SR & FLASH_SR_EOP)
113 FLASH_SR = FLASH_SR_EOP;
114
115 flash_lock_pecr();
116
117 return true;
118}

170bool mem_flash_erase_page(uint32_t address) {
171 bool ret = false;
172 uint8_t num_tries;
173
174 // Check page aligned
175 if (address & 0x7F)
176 return false;
177
178 num_tries = 0;
179 while ((false == ret) && (num_tries < 3)) {
180 flash_unlock();
181 ret = mem_flash_do_page_erase(address);
182 flash_lock();
183
184 num_tries++;
185 }
186
187 if (false == ret) {
188 log_printf("MEM: ERR flash erase fail\n");
189 }
190
191 return ret;
192}

217bool mem_flash_write_word(uint32_t address, uint32_t data) {
218 bool ret = false;
219 uint32_t timer;
220
221 flash_unlock();
222
223 MMIO32(address) = data;
224
225 // wait for completion
226 timer = timers_millis();
227 while (1) {
228 if (true != (FLASH_SR & FLASH_SR_BSY)) {
229 ret = true;
230 break;
231 } else if (timers_millis() - timer < 100) {
232 break;
233 }
234 }
235
236 flash_lock();
237
238 if ((ret) && (FLASH_SR & FLASH_SR_EOP)) {
239 FLASH_SR |= FLASH_SR_EOP;
240 ret = true;
241 } else {
242 ret = false;
243 log_printf("MEM: ERR flash write hp fail\n");
244 }
245
246 return ret;
247}

249void mem_save_reading(int16_t reading) {
250 msg_num++;

255 mem_eeprom_write_word(msg_num_add, msg_num);
256 mem_wipe_readings();

404static _RAM bool mem_flash_do_page_erase(uint32_t address) {
405 bool ret = false;
406 uint32_t counter = rcc_ahb_frequency / 10;
407
408 // erase operation
409 FLASH_PECR |= FLASH_PECR_ERASE | FLASH_PECR_PROG;
410 MMIO32(address) = (uint32_t)0;
411
412 // wait for completion
413 while (1) {
414 if (false == (FLASH_SR & FLASH_SR_BSY)) {
415 ret = true;
416 break;
417 } else if (counter == 0) {
418 break;
419 }
420
421 --counter;
422 }
423
424 if ((ret) && (FLASH_SR & FLASH_SR_EOP)) {
425 // completed without incident
426 FLASH_SR = FLASH_SR_EOP;
427 ret = true;
428 } else {
429 // there was an error
430 FLASH_SR = FLASH_SR_FWWERR | FLASH_SR_PGAERR | FLASH_SR_WRPERR;
431 ret = false;
432 }
433
434 FLASH_PECR &= ~(FLASH_PECR_ERASE | FLASH_PECR_PROG);
435
436 return ret;
437}

446static _RAM bool
447mem_flash_do_write_half_page(uint32_t address,
448 uint32_t data[FLASH_PAGE_SIZE / 2]) {
449 bool ret = false;
450 uint8_t i;
451 uint32_t counter = rcc_ahb_frequency / 10;
452
453 // half-page program operation
454 FLASH_PECR |= FLASH_PECR_PROG | FLASH_PECR_FPRG;
455 for (i = 0; i < 16; i++) {
456 MMIO32(address) = data[i]; // the actual address written is unimportant
457 // as these words will be queued
458 }
459
460 // wait for completion
461 while (1) {
462 if (true != (FLASH_SR & FLASH_SR_BSY)) {
463 ret = true;
464 break;
465 } else if (counter == 0) {
466 break;
467 }
468 --counter;
469 }
470
471 if ((ret) && (FLASH_SR & FLASH_SR_EOP)) {
472 // completed without incident
473 FLASH_SR = FLASH_SR_EOP;
474 ret = true;
475 } else {
476 // there was an error
477 FLASH_SR = FLASH_SR_FWWERR | FLASH_SR_NOTZEROERR | FLASH_SR_PGAERR |
478 FLASH_SR_WRPERR;
479 ret = false;
480 }
481
482 FLASH_PECR &= ~(FLASH_PECR_PROG | FLASH_PECR_FPRG);
483
484 return ret;
