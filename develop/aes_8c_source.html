126// The round constant word array, Rcon[i], contains the values given by
127// x to the power (i-1) being powers of x (x is denoted as {02}) in the field
128// GF(2^8)
129static const uint8_t Rcon[11] = {0x8d, 0x01, 0x02, 0x04, 0x08, 0x10,

153// This function produces Nb(Nr+1) round keys. The round keys are used in each
154// round to decrypt the states.
155static void KeyExpansion(uint8_t* RoundKey, const uint8_t* Key) {
156 unsigned i, j, k;
157 uint8_t tempa[4]; // Used for the column/row operations
158
159 // The first round key is the key itself.
160 for (i = 0; i < Nk; ++i) {
161 RoundKey[(i * 4) + 0] = Key[(i * 4) + 0];
162 RoundKey[(i * 4) + 1] = Key[(i * 4) + 1];
163 RoundKey[(i * 4) + 2] = Key[(i * 4) + 2];
164 RoundKey[(i * 4) + 3] = Key[(i * 4) + 3];
165 }
166
167 // All other round keys are found from the previous round keys.
168 for (i = Nk; i < Nb * (Nr + 1); ++i) {
169 {
170 k = (i - 1) * 4;
171 tempa[0] = RoundKey[k + 0];
172 tempa[1] = RoundKey[k + 1];
173 tempa[2] = RoundKey[k + 2];
174 tempa[3] = RoundKey[k + 3];
175 }
176
177 if (i % Nk == 0) {
178 // This function shifts the 4 bytes in a word to the left once.
179 // [a0,a1,a2,a3] becomes [a1,a2,a3,a0]
180
181 // Function RotWord()
182 {
183 const uint8_t u8tmp = tempa[0];
184 tempa[0] = tempa[1];
185 tempa[1] = tempa[2];
186 tempa[2] = tempa[3];
187 tempa[3] = u8tmp;
188 }
189
190 // SubWord() is a function that takes a four-byte input word and
191 // applies the S-box to each of the four bytes to produce an output
192 // word.
193
194 // Function Subword()
195 {
196 tempa[0] = getSBoxValue(tempa[0]);
197 tempa[1] = getSBoxValue(tempa[1]);
198 tempa[2] = getSBoxValue(tempa[2]);
199 tempa[3] = getSBoxValue(tempa[3]);
200 }
201
202 tempa[0] = tempa[0] ^ Rcon[i / Nk];
203 }
204
205 j = i * 4;
206 k = (i - Nk) * 4;
207 RoundKey[j + 0] = RoundKey[k + 0] ^ tempa[0];
208 RoundKey[j + 1] = RoundKey[k + 1] ^ tempa[1];
209 RoundKey[j + 2] = RoundKey[k + 2] ^ tempa[2];
210 RoundKey[j + 3] = RoundKey[k + 3] ^ tempa[3];
211 }
212}

214// This function adds the round key to state.
215// The round key is added to the state by an XOR function.
216static void AddRoundKey(uint8_t round, state_t* state,
217 const uint8_t* RoundKey) {
218 uint8_t i, j;
219 for (i = 0; i < 4; ++i) {
220 for (j = 0; j < 4; ++j) {
221 (*state)[i][j] ^= RoundKey[(round * Nb * 4) + (i * Nb) + j];
222 }
223 }
224}

226// The SubBytes Function Substitutes the values in the
227// state matrix with values in an S-box.
228static void SubBytes(state_t* state) {
229 uint8_t i, j;
230 for (i = 0; i < 4; ++i) {
231 for (j = 0; j < 4; ++j) {
232 (*state)[j][i] = getSBoxValue((*state)[j][i]);
233 }
234 }
235}

237// The ShiftRows() function shifts the rows in the state to the left.
238// Each row is shifted with different offset.
239// Offset = Row number. So the first row is not shifted.
240static void ShiftRows(state_t* state) {
241 uint8_t temp;
242
243 // Rotate first row 1 columns to left
244 temp = (*state)[0][1];
245 (*state)[0][1] = (*state)[1][1];
246 (*state)[1][1] = (*state)[2][1];
247 (*state)[2][1] = (*state)[3][1];
248 (*state)[3][1] = temp;
249
250 // Rotate second row 2 columns to left
251 temp = (*state)[0][2];
252 (*state)[0][2] = (*state)[2][2];
253 (*state)[2][2] = temp;
254
255 temp = (*state)[1][2];
256 (*state)[1][2] = (*state)[3][2];
257 (*state)[3][2] = temp;
258
259 // Rotate third row 3 columns to left
260 temp = (*state)[0][3];
261 (*state)[0][3] = (*state)[3][3];
262 (*state)[3][3] = (*state)[2][3];
263 (*state)[2][3] = (*state)[1][3];
264 (*state)[1][3] = temp;
265}

267static uint8_t xtime(uint8_t x) {
268 return ((x << 1) ^ (((x >> 7) & 1) * 0x1b));
269}
270
271// MixColumns function mixes the columns of the state matrix
272static void MixColumns(state_t* state) {
273 uint8_t i;
274 uint8_t Tmp, Tm, t;
275 for (i = 0; i < 4; ++i) {
276 t = (*state)[i][0];
277 Tmp = (*state)[i][0] ^ (*state)[i][1] ^ (*state)[i][2] ^ (*state)[i][3];
278 Tm = (*state)[i][0] ^ (*state)[i][1];
279 Tm = xtime(Tm);
280 (*state)[i][0] ^= Tm ^ Tmp;
281 Tm = (*state)[i][1] ^ (*state)[i][2];
282 Tm = xtime(Tm);
283 (*state)[i][1] ^= Tm ^ Tmp;
284 Tm = (*state)[i][2] ^ (*state)[i][3];
285 Tm = xtime(Tm);
286 (*state)[i][2] ^= Tm ^ Tmp;
287 Tm = (*state)[i][3] ^ t;
288 Tm = xtime(Tm);
289 (*state)[i][3] ^= Tm ^ Tmp;
290 }
291}

293// Multiply is used to multiply numbers in the field GF(2^8)
294// Note: The last call to xtime() is unneeded, but often ends up generating a
295// smaller binary
296// The compiler seems to be able to vectorize the operation better this
297// way. See https://github.com/kokke/tiny-AES-c/pull/34
298#if MULTIPLY_AS_A_FUNCTION
299static uint8_t Multiply(uint8_t x, uint8_t y) {
300 return (((y & 1) * x) ^ ((y >> 1 & 1) * xtime(x)) ^
301 ((y >> 2 & 1) * xtime(xtime(x))) ^
302 ((y >> 3 & 1) * xtime(xtime(xtime(x)))) ^
303 ((y >> 4 & 1) *
304 xtime(xtime(xtime(
305 xtime(x)))))); // this last call to xtime() can be omitted
306}
307#else
308#define Multiply(x, y) \
309 (((y & 1) * x) ^ ((y >> 1 & 1) * xtime(x)) ^ \
310 ((y >> 2 & 1) * xtime(xtime(x))) ^ \
311 ((y >> 3 & 1) * xtime(xtime(xtime(x)))) ^ \
312 ((y >> 4 & 1) * xtime(xtime(xtime(xtime(x))))))
313
314#endif

316// MixColumns function mixes the columns of the state matrix.
317// The method used to multiply may be difficult to understand for the
318// inexperienced. Please use the references to gain more information.
319static void InvMixColumns(state_t* state) {
320 int i;
321 uint8_t a, b, c, d;
322 for (i = 0; i < 4; ++i) {
323 a = (*state)[i][0];
324 b = (*state)[i][1];
325 c = (*state)[i][2];
326 d = (*state)[i][3];
327
328 (*state)[i][0] = Multiply(a, 0x0e) ^ Multiply(b, 0x0b) ^
329 Multiply(c, 0x0d) ^ Multiply(d, 0x09);
330 (*state)[i][1] = Multiply(a, 0x09) ^ Multiply(b, 0x0e) ^
331 Multiply(c, 0x0b) ^ Multiply(d, 0x0d);
332 (*state)[i][2] = Multiply(a, 0x0d) ^ Multiply(b, 0x09) ^
333 Multiply(c, 0x0e) ^ Multiply(d, 0x0b);
334 (*state)[i][3] = Multiply(a, 0x0b) ^ Multiply(b, 0x0d) ^
335 Multiply(c, 0x09) ^ Multiply(d, 0x0e);
336 }
337}

377// Cipher is the main function that encrypts the PlainText.
378static void Cipher(state_t* state, const uint8_t* RoundKey) {
379 uint8_t round = 0;
380
381 // Add the First round key to the state before starting the rounds.
382 AddRoundKey(0, state, RoundKey);
383
384 // There will be Nr rounds.
385 // The first Nr-1 rounds are identical.
386 // These Nr rounds are executed in the loop below.
387 // Last one without MixColumns()
388 for (round = 1;; ++round) {
389 SubBytes(state);
390 ShiftRows(state);
391 if (round == Nr) {
392 break;
393 }
394 MixColumns(state);
395 AddRoundKey(round, state, RoundKey);
396 }
397 // Add round key to last round
398 AddRoundKey(Nr, state, RoundKey);
399}
400
401static void InvCipher(state_t* state, const uint8_t* RoundKey) {
402 uint8_t round = 0;
403
404 // Add the First round key to the state before starting the rounds.
405 AddRoundKey(Nr, state, RoundKey);
406
407 // There will be Nr rounds.
408 // The first Nr-1 rounds are identical.
409 // These Nr rounds are executed in the loop below.
410 // Last one without InvMixColumn()
411 for (round = (Nr - 1);; --round) {
412 InvShiftRows(state);
413 InvSubBytes(state);
414 AddRoundKey(round, state, RoundKey);
415 if (round == 0) {
416 break;
417 }
418 InvMixColumns(state);
419 }
420}
