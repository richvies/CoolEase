33void si7051_read_temperature(int16_t* readings, uint8_t num_readings) {
34 // Set resolution of si7051 to 14 bits by writing user register
35 uint8_t data[2] = {SI7051_CMD_WRITE_REG, SI7051_USER_REG_VAL};
36 i2c_transfer7(TEMP_I2C, SI7051_I2C_ADDRESS, data, 2, NULL, 0);
37
38 // Command for making temperature measurement
39 uint8_t cmd[1] = {SI7051_CMD_MEASURE_HOLD};
40
41 // Each reading is made up of two 8 bit values
42 uint8_t reading_msb_lsb[2] = {0, 0};
43
44 for (int i = 0; i < num_readings; i++) {
45 // Transfer measure command and return msb/ lsb of measurement
46 i2c_transfer7(TEMP_I2C, SI7051_I2C_ADDRESS, cmd, 1, reading_msb_lsb, 2);
47
48 // Convert to degrees celcius and store in array
49 uint32_t temp = reading_msb_lsb[0] << 24 | reading_msb_lsb[1] << 16;
50 temp = temp >> 16;
51 temp = temp * 17572;
52 temp = temp / 65536;
53 temp = temp - 4685;
54
55 readings[i] = temp;
56 }
57}
