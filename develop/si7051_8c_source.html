45void si7051_read_temperature(int16_t* readings, uint8_t num_readings) {
46 // Set resolution of si7051 to 14 bits by writing user register
47 uint8_t data[2] = {SI7051_CMD_WRITE_REG, SI7051_USER_REG_VAL};
48 i2c_transfer7(TEMP_I2C, SI7051_I2C_ADDRESS, data, 2, NULL, 0);
49
50 // Command for making temperature measurement
51 uint8_t cmd[1] = {SI7051_CMD_MEASURE_HOLD};
52
53 // Each reading is made up of two 8 bit values
54 uint8_t reading_msb_lsb[2] = {0, 0};
55
56 for (int i = 0; i < num_readings; i++) {
57 // Transfer measure command and return msb/ lsb of measurement
58 i2c_transfer7(TEMP_I2C, SI7051_I2C_ADDRESS, cmd, 1, reading_msb_lsb, 2);
59
60 // Convert to degrees celcius and store in array
61 uint32_t temp = reading_msb_lsb[0] << 24 | reading_msb_lsb[1] << 16;
62 temp = temp >> 16;
63 temp = temp * 17572;
64 temp = temp / 65536;
65 temp = temp - 4685;
66
67 readings[i] = temp;
68 }
69}
