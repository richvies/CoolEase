50 for (int i = 0; i < num_readings; i++) {
51 // Set One Shot bit in config register to start conversion
52 uint8_t config[3] = {TMP112_SEL_CONFIG_REG, TMP112_CONFIG_OS_MSB,
53 TMP112_CONFIG_OS_LSB};
54 i2c_transfer(TEMP_I2C, TMP112_I2C_ADDRESS, config, 3, NULL, 0);
55
56 // Delay 50ms for conversion to complete
57 timers_delay_microseconds(50000);
58
59 // Select temperature register
60 uint8_t select_temp[1] = {TMP112_SEL_TEMP_REG};
61 i2c_transfer(TEMP_I2C, TMP112_I2C_ADDRESS, select_temp, 1, NULL, 0);
62
63 // Each reading is made up of two 8 bit values
64 uint8_t reading_msb_lsb[2] = {0, 0};

69 // Convert to degrees celcius and store in array
70 int32_t temp = reading_msb_lsb[0] << 24 | reading_msb_lsb[1] << 16;
