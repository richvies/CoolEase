61void tmp112_read_temperature(int16_t* readings, uint8_t num_readings) {
62 for (int i = 0; i < num_readings; i++) {
63 // Set One Shot bit in config register to start conversion
64 uint8_t config[3] = {TMP112_SEL_CONFIG_REG, TMP112_CONFIG_OS_MSB,
65 TMP112_CONFIG_OS_LSB};
66 i2c_transfer(TEMP_I2C, TMP112_I2C_ADDRESS, config, 3, NULL, 0);
67
68 // Delay 50ms for conversion to complete
69 timers_delay_microseconds(50000);
70
71 // Select temperature register
72 uint8_t select_temp[1] = {TMP112_SEL_TEMP_REG};
73 i2c_transfer(TEMP_I2C, TMP112_I2C_ADDRESS, select_temp, 1, NULL, 0);
74
75 // Each reading is made up of two 8 bit values
76 uint8_t reading_msb_lsb[2] = {0, 0};
77
78 // Read temperature register and return msb/ lsb of measurement
79 i2c_transfer(TEMP_I2C, TMP112_I2C_ADDRESS, NULL, 0, reading_msb_lsb, 2);
80
81 // Convert to degrees celcius and store in array
82 int32_t temp = reading_msb_lsb[0] << 24 | reading_msb_lsb[1] << 16;
83 temp = temp >> 20;
84
85 temp *= 625;
86 temp /= 100;
87 readings[i] = temp;
88 }
