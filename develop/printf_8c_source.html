61 // internal itoa format
62 static uint32_t _ntoa_format(out_fct_type out, uint32_t value, uint32_t base,
63  uint32_t width, bool negative) {
64  char buf[PRINTF_NTOA_BUFFER_SIZE];
65  uint32_t len = 0;
66 
67  // Digits to char
68  do {
69  const char digit = (char)(value % base);
70  buf[len++] = digit < 10 ? '0' + digit : ('A') + digit - 10;
71  value /= base;
72  } while (value && (len < PRINTF_NTOA_BUFFER_SIZE));
73 
74  // pad leading zeros
75  while ((len < width) && (len < PRINTF_NTOA_BUFFER_SIZE)) {
76  buf[len++] = '0';
77  }
78 
79  // Base specifier
80  if ((base == 16U) && (len < PRINTF_NTOA_BUFFER_SIZE)) {
81  buf[len++] = 'X';
82  } else if ((base == 2U) && (len < PRINTF_NTOA_BUFFER_SIZE)) {
83  buf[len++] = 'b';
84  }
85  if (((base == 16U) || (base == 2U)) && (len < PRINTF_NTOA_BUFFER_SIZE)) {
86  buf[len++] = '0';
87  }
88 
89  if (negative && (len < PRINTF_NTOA_BUFFER_SIZE)) {
90  buf[len++] = '-';
91  }
92 
93  // reverse string and write out
94  uint32_t i = len;
95  while (i) {
96  out(buf[--i]);
97  }
98  return len;
99 }

101 uint32_t fnprintf(out_fct_type out, const char* format, va_list va) {
102  uint32_t width;
103  uint32_t idx = 0U;
104 
105  while (*format) {
106  // format specifier? %[flags][width]
107  if (*format == '%') {
108  // Yes, evaluate it
109  format++;
110  } else {
111  // Just print char
112  out(*format);
113  idx++;
114  format++;
115  continue;
116  }
117 
118  // evaluate width field
119  width = 0U;
120  if (_is_digit(*format)) {
121  width = _atoi(&format);
122  }
123 
124  // evaluate specifier
125  switch (*format) {
126  case 'd':
127  case 'i':
128  case 'u':
129  case 'x':
130  case 'X':
131  case 'o':
132  case 'b': {
133  // set the base
134  uint32_t base;
135  if (*format == 'x' || *format == 'X') {
136  base = 16U;
137  } else if (*format == 'o') {
138  base = 8U;
139  } else if (*format == 'b') {
140  base = 2U;
141  } else {
142  base = 10U;
143  }
144 
145  // convert integer
146  if ((*format == 'i') || (*format == 'd')) {
147  int32_t val = (int32_t)va_arg(va, int);
148  idx += _ntoa_format(out, (val < 0) ? (0 - val) : val, base,
149  width, (val < 0) ? true : false);
150  } else {
151  idx += _ntoa_format(out, (uint32_t)va_arg(va, int), base, width,
152  false);
153  }
154  format++;
155  break;
156  }
157 
158  case 'c': {
159  out((char)va_arg(va, int));
160  idx++;
161  format++;
162  break;
163  }
164 
165  case 's': {
166  const char* p = va_arg(va, char*);
167  uint32_t l = _strnlen_s(p, PRINTF_MAX_STR_LEN);
168 
169  // string output
170  while ((*p != 0) && (l--)) {
171  out(*(p++));
172  idx++;
173  }
174  format++;
175  break;
176  }
177 
178  case 'p': {
179  width = sizeof(void*) * 2U;
180  idx += _ntoa_format(out, (uint32_t)((uintptr_t)va_arg(va, void*)),
181  16U, width, false);
182  format++;
183  break;
184  }
185 
186  case '%': {
187  out('%');
188  idx++;
189  format++;
190  break;
191  }
192 
193  default: {
194  out(*format);
195  idx++;
196  format++;
197  break;
198  }
199  }
200  }
201  // return written chars
202  return idx;
203 }
