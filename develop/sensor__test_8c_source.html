202void test_sensor_rf_vs_temp_cal(void) {
203 test_init("test_sensor_rf_vs_temp_cal()");
204
205 /*////////////////////////*/
206 // Get Average Temperature
207 /*////////////////////////*/
208 uint8_t max_readings = 4;
209 int16_t readings[4] = {22222, 22222, 22222, 22222};
210 tmp112_init();
211 tmp112_read_temperature(readings, max_readings);
212 tmp112_end();
213 int32_t sum = 0;
214 uint8_t num_readings = 0;
215 int16_t temp_avg = 22222;
216 for (int i = 0; i < max_readings; i++) {
217 if (readings[i] != 22222) {
218 sum += readings[i];
219 num_readings++;
220 }
221 }
222 // Only calc avg if num_readings > 0
223 if (num_readings) {
224 temp_avg = sum / num_readings;
225 }
226 serial_printf("Temp: %i\n", temp_avg);
227
228 /*////////////////////////*/
229 // Assemble Packet
230 /*////////////////////////*/
231 rfm_packet_t packet;
232 packet.data.temperature = temp_avg;
233 packet.data.msg_number = 0;
234 packet.data.device_number = 0xAD7503BF;
235
236 /*////////////////////////*/
237 // Send Packet
238 /*////////////////////////*/
239 rfm_init();
240 rfm_config_for_lora(RFM_BW_125KHZ, RFM_CODING_RATE_4_5,
241 RFM_SPREADING_FACTOR_128CPS, true, 0);
242 rfm_transmit_packet(packet);
243 serial_printf("Sent\n");
244
245 /*////////////////////////*/
246 // 1 Second High Pulse
247 /*////////////////////////*/
248 rfm_set_tx_continuous();
249 timers_delay_milliseconds(1000);
250 rfm_end();
251
252 /*////////////////////////*/
253 // Enter Standby for 30 seconds
254 // Resets in RTC ISR
255 /*////////////////////////*/
256 timers_rtc_init();
257 timers_set_wakeup_time(30);
258 timers_enable_wut_interrupt();
259 set_gpio_for_standby();
260 timers_enter_standby();
261}
