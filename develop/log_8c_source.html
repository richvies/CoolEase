269static void _putchar_spf(char character) {
270 bool done = false;
271
272 // send immediatly if tx buffer empty
273 if ((spf_tx_head == spf_tx_tail) &&
274 usart_get_flag(SPF_USART, USART_ISR_TXE)) {
275 usart_send(SPF_USART, character);
276 done = true;
277 } else {
278 while (!done) {
279 cm_disable_interrupts();
280
281 uint8_t i = (spf_tx_head + 1) % SPF_BUFFER_SIZE;
282
283 if (i == spf_tx_tail) {
284 done = false;
285 cm_enable_interrupts();
286 timers_delay_milliseconds(1);
287 } else {
288
289 spf_tx_buf[spf_tx_head] = character;
290
291 spf_tx_head = i;
292
293 usart_enable_tx_interrupt(SPF_USART);
294
295 done = true;
296 }
297 cm_enable_interrupts();
298 }
299 }

324SPF_ISR() {
325 // Received data from serial monitor
326 // Read data (clears flag automatically)
327 if (usart_get_flag(SPF_USART, USART_ISR_RXNE)) {
328 if (((spf_rx_head + 1) % SPF_BUFFER_SIZE) == spf_rx_tail) {
329 // Read overflow
330 usart_recv(SPF_USART);
331 } else {
332 spf_rx_buf[spf_rx_head] = usart_recv(SPF_USART);
333
334 // if (spf_rx_buf[spf_rx_head] == '\n')
335 // {
336 // spf_rx_buf[spf_rx_head + 1] = '\0'
337 // }
338
339 spf_rx_head = (spf_rx_head + 1) % SPF_BUFFER_SIZE;
340 }
341 }
342
343 // Transmit buffer empty
344 // Fill it (clears flag automatically)
345 if (usart_get_flag(SPF_USART, USART_ISR_TXE)) {
346 // Fill it if data waiting
347 if ((SPF_BUFFER_SIZE + spf_tx_head - spf_tx_tail) % SPF_BUFFER_SIZE) {
348 usart_send(SPF_USART, spf_tx_buf[spf_tx_tail]);
349 spf_tx_tail = (spf_tx_tail + 1) % SPF_BUFFER_SIZE;
350 }
351 // Otherwise tranfer is done, disable interrupt (prevent irq firing
352 // constantly waiting for TX Data Reg to be filled)
353 else {
354 usart_disable_tx_interrupt(SPF_USART);
355 }
356 }
357}
