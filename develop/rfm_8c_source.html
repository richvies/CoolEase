259void rfm_set_power(int8_t power, uint8_t ramp_time) {
260 if (power > 20)
261 power = 20;
262 // else if( power < -3 )
263 // power = -3;
264
265 if (ramp_time > 0x0F)
266 ramp_time = 0x0F;
267
268 // Pout = 2 + OutputPower (+3dBm if DAC enabled)
269 spi_write_single(RFM_REG_4D_PA_DAC,
270 (spi_read_single(RFM_REG_4D_PA_DAC) & ~RFM_PA_DAC_MASK) |
271 RFM_PA_DAC_DISABLE);
272
273 // Set the MaxPower register to 0x7 => MaxPower = 10.8 + 0.6 * 7 = 15dBm
274 // So Pout = 17 - (15 - power) = 2 + power
275 spi_write_single(RFM_REG_09_PA_CONFIG,
276 RFM_PA_SELECT | RFM_MAX_POWER | (power - 2));
277
278 // Set ramp time
279 spi_write_single(RFM_REG_0A_PA_RAMP,
280 (spi_read_single(RFM_REG_0A_PA_RAMP) & ~RFM_PA_RAMP_MASK) |
281 ramp_time);
282}

296void rfm_start_listening(void) {
297 // Go to standby mode
298 set_standby_mode();
299
300 // Set buffer offset to 0
301 clear_buffer();
302
303 // Set RX Done IRQ on IO0
304 set_dio_irq(RFM_IO_0_IRQ_RX_DONE | RFM_IO_1_IRQ_RX_TIMEOUT |
305 RFM_IO_2_IRQ_FHSS_CHANGE | RFM_IO_3_IRQ_CRC_ERROR,
306 RFM_IO_4_IRQ_CAD_DETECTED | RFM_IO_5_IRQ_MODE_READY);
307
308 // Unmask interrupt for RX Done
309 mask_irq(RFM_IRQ_ALL);
310 unmask_irq(RFM_RX_TIMEOUT_MASK | RFM_RX_DONE_MASK | RFM_VALID_HEADER_MASK |
311 RFM_PAYLOAD_CRC_ERROR_MASK);
312 clear_irq(RFM_IRQ_ALL);
313
314 // Enable interrupt on MCU for RX done GPIO_RF_IO_0
315 exti_reset_request(RFM_IO_0_EXTI);
316 exti_select_source(RFM_IO_0_EXTI, RFM_IO_0_PORT);
317 exti_set_trigger(RFM_IO_0_EXTI, EXTI_TRIGGER_RISING);
318 exti_enable_request(RFM_IO_0_EXTI);
319
320 nvic_enable_irq(RFM_IO_0_NVIC);
321 nvic_set_priority(RFM_IO_0_NVIC, IRQ_PRIORITY_RFM);
322
323 // Start listening
324 set_rx_mode();
325}

395 RFM_IO_4_IRQ_CAD_DETECTED | RFM_IO_5_IRQ_MODE_READY);
396 mask_irq(RFM_IRQ_ALL);

400 // Write packet length

403 // spi_write_single(RFM_REG_22_PAYLOAD_LENGTH, packet.length);
404
405 // Write packet data
406 spi_write_burst(RFM_REG_00_FIFO, packet.data.buffer, RFM_PACKET_LENGTH);
407 // spi_write_burst(RFM_REG_00_FIFO, packet.data, packet.length);
408 // log_printf("SPI Pointer: %02x : %02x\n", RFM_REG_0D_FIFO_ADDR_PTR,
409 // spi_read_single(RFM_REG_0D_FIFO_ADDR_PTR));
410
411 // Wait for clear channel
412
413 // About 50ms to send packet currently
414 // uint16_t start = timers_millis();
415
416 // Enter TX state
417 set_tx_mode();
418
419 // wait_rf_io_0_high();

421 // log_printf("Transmit Time: %u ms\n", (uint16_t)(end - start));

425 TIMEOUT(RFM_TX_TIMEOUT, "RFM TX", 0, gpio_get(RFM_IO_0_PORT, RFM_IO_0),
