97 void batt_init(void) {
98  // Disable Interrupt
99  nvic_disable_irq(NVIC_DMA1_CHANNEL1_IRQ);
100 
101  // Config Vrefint
102  rcc_periph_clock_enable(RCC_SYSCFG);
103 
104  // Start/ wait for voltage reference, enable buffer to adc
105  SYSCFG_CFGR3 |= SYSCFG_CFGR3_EN_VREFINT;
106  while (!(SYSCFG_CFGR3 & SYSCFG_CFGR3_VREFINT_RDYF)) {
107  }
108  SYSCFG_CFGR3 |= SYSCFG_CFGR3_ENBUF_VREFINT_ADC;
109 
110  // Enable adc clock and calibrate
111  rcc_periph_clock_enable(RCC_ADC1);
112  rcc_periph_reset_pulse(RST_ADC1);
113 
114  // Clock config
115  uint32_t adc_freq;
116  if (sys_clk == RCC_HSI16) {
117  // Set clock to HSI clk
118  ADC_CFGR2(ADC1) &= ~ADC_CFGR2_CKMODE;
119 
120  // Set prescaler 32
121  ADC_CCR(ADC1) &= ~ADC_CCR_PRESC;
122  ADC_CCR(ADC1) |= ADC_CCR_PRESC_DIV32;
123 
124  adc_freq = 16000000 / 32;
125  } else {
126  // Set clock to APB clk / 4
127  ADC_CFGR2(ADC1) &= ~ADC_CFGR2_CKMODE;
128  ADC_CFGR2(ADC1) |= ADC_CFGR2_CKMODE_PCLK_DIV4;
129  adc_freq = rcc_apb2_frequency / 4;
130  }
131 
132  // Enable low frequency below 2.8MHz, pg.297 of ref
133  if (adc_freq < 2800000) {
134  ADC_CCR(ADC1) |= ADC_CCR_LFMEN;
135  } else {
136  ADC_CCR(ADC1) &= ~ADC_CCR_LFMEN;
137  }
138 
139  // Power off & calibrate
140  adc_power_off(ADC1);
141  adc_calibrate(ADC1);
142 
143  // Highest sampling time. 160.5 for l051 (239.5 for l052)
144  ADC_SMPR1(ADC1) |= ADC_SMPR_SMP_160DOT5CYC;
145 
146  // Reverse scan direction so that VREF is always first conversion
147  ADC_CFGR1(ADC1) |= ADC_CFGR1_SCANDIR;
148 
149  // Enable voltage reference
150  adc_enable_vrefint();
151 
152  // Turn on ADC
153  adc_power_on(ADC1);
154  timers_delay_microseconds(1000);
155 
156  // Enable analog pins
157  rcc_periph_clock_enable(RCC_GPIOA);
158  gpio_mode_setup(BATT_SENS_PORT, GPIO_MODE_ANALOG, GPIO_PUPD_NONE,
159  BATT_SENS);
160 
161  // Set channels to convert
162  ADC_CHSELR(ADC1) |= (1 << ADC_CHANNEL_VREF);
163  ADC_CHSELR(ADC1) |= (1 << 0);
164 
165 #ifdef _HUB
166  gpio_mode_setup(PWR_SENS_PORT, GPIO_MODE_ANALOG, GPIO_PUPD_NONE, PWR_SENS);
167  ADC_CHSELR(ADC1) |= (1 << 1);
168 #endif
169 
170  log_printf("Batt Init\n");
171 }

254 
255  ADC_CR(ADC1) |= ADC_CR_ADSTP;
256  while (!(ADC_CR(ADC1) & ADC_CR_ADSTP)) {
257  if (timers_millis() - timer > 2000) {

259  return;

263

274  }

405 void dma1_channel1_isr(void) {
406  static uint32_t timer = 0;
407  static batt_state_t last_state = BATT_INIT;
408  static batt_state_t curr_state = BATT_INIT;
409 
410  dma_clear_interrupt_flags(DMA1, DMA_CHANNEL1,
411  DMA_TEIF | DMA_TCIF | DMA_HTIF | DMA_GIF);
412 
413  batt_calculate_voltages();
414 
415  curr_state = (batt_get_pwr_voltage() >= batt_get_batt_voltage())
416  ? BATT_PLUGGED_IN
417  : BATT_PLUGGED_OUT;
418 
419  // if state changed since last check
420  if (curr_state != last_state) {
421  timer = timers_millis();
422  } else {
423  // update state after lag
424  if ((state != curr_state) && (timers_millis() - timer > BATT_LAG_MS)) {
425  state = curr_state;
426  }
427  }
428 
429  last_state = curr_state;
