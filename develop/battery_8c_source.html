75void batt_init(void) {
76 // Disable Interrupt
77 nvic_disable_irq(NVIC_DMA1_CHANNEL1_IRQ);
78
79 // Config Vrefint
80 rcc_periph_clock_enable(RCC_SYSCFG);
81
82 // Start/ wait for voltage reference, enable buffer to adc
83 SYSCFG_CFGR3 |= SYSCFG_CFGR3_EN_VREFINT;
84 while (!(SYSCFG_CFGR3 & SYSCFG_CFGR3_VREFINT_RDYF)) {
85 }
86 SYSCFG_CFGR3 |= SYSCFG_CFGR3_ENBUF_VREFINT_ADC;
87
88 // Enable adc clock and calibrate
89 rcc_periph_clock_enable(RCC_ADC1);
90 rcc_periph_reset_pulse(RST_ADC1);
91
92 // Clock config
93 uint32_t adc_freq;
94 if (sys_clk == RCC_HSI16) {
95 // Set clock to HSI clk
96 ADC_CFGR2(ADC1) &= ~ADC_CFGR2_CKMODE;
97
98 // Set prescaler 32
99 ADC_CCR(ADC1) &= ~ADC_CCR_PRESC;
100 ADC_CCR(ADC1) |= ADC_CCR_PRESC_DIV32;
101
102 adc_freq = 16000000 / 32;
103 } else {
104 // Set clock to APB clk / 4
105 ADC_CFGR2(ADC1) &= ~ADC_CFGR2_CKMODE;
106 ADC_CFGR2(ADC1) |= ADC_CFGR2_CKMODE_PCLK_DIV4;
107 adc_freq = rcc_apb2_frequency / 4;
108 }
109
110 // Enable low frequency below 2.8MHz, pg.297 of ref
111 if (adc_freq < 2800000) {
112 ADC_CCR(ADC1) |= ADC_CCR_LFMEN;
113 } else {
114 ADC_CCR(ADC1) &= ~ADC_CCR_LFMEN;
115 }
116
117 // Power off & calibrate
118 adc_power_off(ADC1);
119 adc_calibrate(ADC1);
120
121 // Highest sampling time. 160.5 for l051 (239.5 for l052)
122 ADC_SMPR1(ADC1) |= ADC_SMPR_SMP_160DOT5CYC;
123
124 // Reverse scan direction so that VREF is always first conversion
125 ADC_CFGR1(ADC1) |= ADC_CFGR1_SCANDIR;
126
127 // Enable voltage reference
128 adc_enable_vrefint();
129
130 // Turn on ADC
131 adc_power_on(ADC1);
132 timers_delay_microseconds(1000);
133
134 // Enable analog pins
135 rcc_periph_clock_enable(RCC_GPIOA);
136 gpio_mode_setup(BATT_SENS_PORT, GPIO_MODE_ANALOG, GPIO_PUPD_NONE,
137 BATT_SENS);
138
139 // Set channels to convert
140 ADC_CHSELR(ADC1) |= (1 << ADC_CHANNEL_VREF);
141 ADC_CHSELR(ADC1) |= (1 << 0);
142
143#ifdef COOLEASE_DEVICE_HUB
144 gpio_mode_setup(PWR_SENS_PORT, GPIO_MODE_ANALOG, GPIO_PUPD_NONE, PWR_SENS);
145 ADC_CHSELR(ADC1) |= (1 << 1);
146#endif
147
148 log_printf("Batt Init\n");
149}

184void batt_set_low_power_run(void) {
185 rcc_periph_clock_enable(RCC_PWR);
186
187 // Set LPRUN & LPSDSR bits in PWR_CR register
188 PWR_CR |= PWR_CR_LPSDSR;
189 PWR_CR |= PWR_CR_LPRUN;
190
191 // Enable MSI Osc 2.097Mhz
192 rcc_osc_on(RCC_MSI);
193 rcc_wait_for_osc_ready(RCC_MSI);
194
195 // Set MSI to 65.536kHz
196 rcc_set_msi_range(0);
197
198 // Set prescalers for AHB, APB1, APB2
199 rcc_set_hpre(RCC_CFGR_HPRE_NODIV); // AHB -> 65.536kHz
200 rcc_set_ppre1(RCC_CFGR_PPRE1_NODIV); // APB1 -> 65.536kHz
201 rcc_set_ppre2(RCC_CFGR_PPRE2_NODIV); // APB2 -> 65.536kHz
202
203 // Set flash, 65.536kHz -> 0 waitstates
204 flash_set_ws(FLASH_ACR_LATENCY_0WS);
205
206 // Set Peripheral Clock Frequencies used
207 rcc_ahb_frequency = 65536;
208 rcc_apb1_frequency = 65536;
209 rcc_apb2_frequency = 65536;
210
211 rcc_periph_clock_disable(RCC_PWR);
212}

214void batt_calculate_voltages(void) {
215 // Calculate batt_voltages
216 for (uint8_t i = 0; i < NUM_VOLTAGES; i++) {
217 batt_voltages[i] = ((uint32_t)300 * ST_VREFINT_CAL * adc_vals[i + 1]) /
218 (adc_vals[0] * 4095);
219 }
220
221// For Hub : Measured voltage is half of actual
222#ifdef COOLEASE_DEVICE_HUB
223 for (uint8_t i = 0; i < NUM_VOLTAGES; i++) {
224 batt_voltages[i] = batt_voltages[i] * 2;
225 }
226#endif
227}

229void batt_update_voltages(void) {
230 static uint32_t timer;
231 timer = timers_millis();
232
233 ADC_CR(ADC1) |= ADC_CR_ADSTP;
234 while (!(ADC_CR(ADC1) & ADC_CR_ADSTP)) {
235 if (timers_millis() - timer > 2000) {
236 log_printf("ERR: ADC Stop Timeout\n");
237 return;
238 }
239 }
240 adc_set_single_conversion_mode(ADC1);
241
242 // Start conversions
243 adc_start_conversion_regular(ADC1);
244 for (uint8_t i = 0; i < NUM_VOLTAGES + 1; i++) {
245 while (!adc_eoc(ADC1)) {
246 if (timers_millis() - timer > 5000) {
247 log_printf("ERR: ADC Conv Timeout\n");
248 return;
249 }
250 }
251 adc_vals[i] = adc_read_regular(ADC1);
252 }
253
254 batt_calculate_voltages();
255}

372void dma1_channel1_isr(void) {
373 static uint32_t timer = 0;
374 static batt_state_t last_state = BATT_INIT;
375 static batt_state_t curr_state = BATT_INIT;
376
377 dma_clear_interrupt_flags(DMA1, DMA_CHANNEL1,
378 DMA_TEIF | DMA_TCIF | DMA_HTIF | DMA_GIF);
379
380 batt_calculate_voltages();
381
382 curr_state = (batt_get_pwr_voltage() >= batt_get_batt_voltage())
383 ? BATT_PLUGGED_IN
384 : BATT_PLUGGED_OUT;
385
386 // if state changed since last check
387 if (curr_state != last_state) {
388 timer = timers_millis();
389 } else {
390 // update state after lag
391 if ((state != curr_state) && (timers_millis() - timer > BATT_LAG_MS)) {
392 state = curr_state;
393 }
394 }
395
396 last_state = curr_state;
397}
