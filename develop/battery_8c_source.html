97void batt_init(void) {
98 // Disable Interrupt
99 nvic_disable_irq(NVIC_DMA1_CHANNEL1_IRQ);
100
101 // Config Vrefint
102 rcc_periph_clock_enable(RCC_SYSCFG);
103
104 // Start/ wait for voltage reference, enable buffer to adc
105 SYSCFG_CFGR3 |= SYSCFG_CFGR3_EN_VREFINT;
106 while (!(SYSCFG_CFGR3 & SYSCFG_CFGR3_VREFINT_RDYF)) {
107 }
108 SYSCFG_CFGR3 |= SYSCFG_CFGR3_ENBUF_VREFINT_ADC;
109
110 // Enable adc clock and calibrate
111 rcc_periph_clock_enable(RCC_ADC1);
112 rcc_periph_reset_pulse(RST_ADC1);
113
114 // Clock config
115 uint32_t adc_freq;
116 if (sys_clk == RCC_HSI16) {
117 // Set clock to HSI clk
118 ADC_CFGR2(ADC1) &= ~ADC_CFGR2_CKMODE;
119
120 // Set prescaler 32
121 ADC_CCR(ADC1) &= ~ADC_CCR_PRESC;
122 ADC_CCR(ADC1) |= ADC_CCR_PRESC_DIV32;
123
124 adc_freq = 16000000 / 32;
125 } else {
126 // Set clock to APB clk / 4
127 ADC_CFGR2(ADC1) &= ~ADC_CFGR2_CKMODE;
128 ADC_CFGR2(ADC1) |= ADC_CFGR2_CKMODE_PCLK_DIV4;
129 adc_freq = rcc_apb2_frequency / 4;
130 }
131
132 // Enable low frequency below 2.8MHz, pg.297 of ref
133 if (adc_freq < 2800000) {
134 ADC_CCR(ADC1) |= ADC_CCR_LFMEN;
135 } else {
136 ADC_CCR(ADC1) &= ~ADC_CCR_LFMEN;
137 }
138
139 // Power off & calibrate
140 adc_power_off(ADC1);
141 adc_calibrate(ADC1);
142
143 // Highest sampling time. 160.5 for l051 (239.5 for l052)
144 ADC_SMPR1(ADC1) |= ADC_SMPR_SMP_160DOT5CYC;
145
146 // Reverse scan direction so that VREF is always first conversion
147 ADC_CFGR1(ADC1) |= ADC_CFGR1_SCANDIR;
148
149 // Enable voltage reference
150 adc_enable_vrefint();
151
152 // Turn on ADC
153 adc_power_on(ADC1);
154 timers_delay_microseconds(1000);
155
156 // Enable analog pins
157 rcc_periph_clock_enable(RCC_GPIOA);
158 gpio_mode_setup(BATT_SENS_PORT, GPIO_MODE_ANALOG, GPIO_PUPD_NONE,
159 BATT_SENS);
160
161 // Set channels to convert
162 ADC_CHSELR(ADC1) |= (1 << ADC_CHANNEL_VREF);
163 ADC_CHSELR(ADC1) |= (1 << 0);
164
165#ifdef _HUB
166 gpio_mode_setup(PWR_SENS_PORT, GPIO_MODE_ANALOG, GPIO_PUPD_NONE, PWR_SENS);
167 ADC_CHSELR(ADC1) |= (1 << 1);
168#endif
169
170 log_printf("Batt Init\n");
171}

173void batt_end(void) {
174 // Disable Interrupts
175 nvic_disable_irq(NVIC_DMA1_CHANNEL1_IRQ);
176
177 adc_power_off(ADC1);
178 adc_disable_vrefint();
179
180 // Turn off ADC Regulator
181 ADC_CR(ADC1) &= ~ADC_CR_ADVREGEN;
182

184 rcc_periph_clock_disable(RCC_ADC1);
185

187 SYSCFG_CFGR3 &= ~SYSCFG_CFGR3_ENBUF_VREFINT_ADC;

206void batt_set_low_power_run(void) {
207 rcc_periph_clock_enable(RCC_PWR);
208
209 // Set LPRUN & LPSDSR bits in PWR_CR register
210 PWR_CR |= PWR_CR_LPSDSR;
211 PWR_CR |= PWR_CR_LPRUN;
212
213 // Enable MSI Osc 2.097Mhz
214 rcc_osc_on(RCC_MSI);
215 rcc_wait_for_osc_ready(RCC_MSI);
216
217 // Set MSI to 65.536kHz
218 rcc_set_msi_range(0);
219
220 // Set prescalers for AHB, APB1, APB2
221 rcc_set_hpre(RCC_CFGR_HPRE_NODIV); // AHB -> 65.536kHz
222 rcc_set_ppre1(RCC_CFGR_PPRE1_NODIV); // APB1 -> 65.536kHz
223 rcc_set_ppre2(RCC_CFGR_PPRE2_NODIV); // APB2 -> 65.536kHz
224
225 // Set flash, 65.536kHz -> 0 waitstates
226 flash_set_ws(FLASH_ACR_LATENCY_0WS);
227
228 // Set Peripheral Clock Frequencies used
229 rcc_ahb_frequency = 65536;
230 rcc_apb1_frequency = 65536;
231 rcc_apb2_frequency = 65536;
232
233 rcc_periph_clock_disable(RCC_PWR);
234}

236void batt_calculate_voltages(void) {
237 // Calculate batt_voltages
238 for (uint8_t i = 0; i < NUM_VOLTAGES; i++) {
239 batt_voltages[i] = ((uint32_t)300 * ST_VREFINT_CAL * adc_vals[i + 1]) /
240 (adc_vals[0] * 4095);
241 }
242
243// For Hub : Measured voltage is half of actual
244#ifdef _HUB
245 for (uint8_t i = 0; i < NUM_VOLTAGES; i++) {
246 batt_voltages[i] = batt_voltages[i] * 2;
247 }
248#endif
249}

251void batt_update_voltages(void) {
252 static uint32_t timer;
253 timer = timers_millis();
254
255 ADC_CR(ADC1) |= ADC_CR_ADSTP;
256 while (!(ADC_CR(ADC1) & ADC_CR_ADSTP)) {
257 if (timers_millis() - timer > 2000) {
258 log_printf("ERR: ADC Stop Timeout\n");
259 return;
260 }
261 }
262 adc_set_single_conversion_mode(ADC1);
263
264 // Start conversions
265 adc_start_conversion_regular(ADC1);
266 for (uint8_t i = 0; i < NUM_VOLTAGES + 1; i++) {
267 while (!adc_eoc(ADC1)) {
268 if (timers_millis() - timer > 5000) {
269 log_printf("ERR: ADC Conv Timeout\n");
270 return;
271 }
272 }
273 adc_vals[i] = adc_read_regular(ADC1);
274 }
275
276 batt_calculate_voltages();
277}

279void batt_enable_interrupt(void) {
280 ADC_CR(ADC1) |= ADC_CR_ADSTP;
281 while (!(ADC_CR(ADC1) & ADC_CR_ADSTP)) {
282 }
283 adc_set_continuous_conversion_mode(ADC1);
284
285 adc_enable_dma(ADC1);
286 adc_enable_dma_circular_mode(ADC1);
287
288 // Setup DMA
289 rcc_periph_clock_enable(RCC_DMA);
290 dma_channel_reset(DMA1, 1);
291
292 dma_enable_circular_mode(DMA1, 1);
293
294 dma_set_read_from_peripheral(DMA1, 1);
295 dma_set_number_of_data(DMA1, 1, NUM_VOLTAGES + 1);
296 dma_set_priority(DMA1, 1, DMA_CCR_PL_VERY_HIGH);
297
298 dma_set_peripheral_address(DMA1, 1, (uint32_t)&ADC_DR(ADC1));
299 dma_set_peripheral_size(DMA1, 1, DMA_CCR_PSIZE_32BIT);
300 dma_disable_peripheral_increment_mode(DMA1, 1);
301
302 dma_set_memory_address(DMA1, 1, (uint32_t)adc_vals);
303 dma_set_memory_size(DMA1, 1, DMA_CCR_MSIZE_16BIT);
304 dma_enable_memory_increment_mode(DMA1, 1);
305
306 dma_enable_transfer_complete_interrupt(DMA1, DMA_CHANNEL1);
307
308 dma_enable_channel(DMA1, 1);
309
310 // Enable interrupt. Low priority
311 nvic_enable_irq(NVIC_DMA1_CHANNEL1_IRQ);
312 nvic_set_priority(NVIC_DMA1_CHANNEL1_IRQ, IRQ_PRIORITY_BATT);
313
314 // Start conversions
315 adc_start_conversion_regular(ADC1);
316}

405void dma1_channel1_isr(void) {
406 static uint32_t timer = 0;
407 static batt_state_t last_state = BATT_INIT;
408 static batt_state_t curr_state = BATT_INIT;
409
410 dma_clear_interrupt_flags(DMA1, DMA_CHANNEL1,
411 DMA_TEIF | DMA_TCIF | DMA_HTIF | DMA_GIF);
412
413 batt_calculate_voltages();
414
415 curr_state = (batt_get_pwr_voltage() >= batt_get_batt_voltage())
416 ? BATT_PLUGGED_IN
417 : BATT_PLUGGED_OUT;
418
419 // if state changed since last check
420 if (curr_state != last_state) {
421 timer = timers_millis();
422 } else {
423 // update state after lag
424 if ((state != curr_state) && (timers_millis() - timer > BATT_LAG_MS)) {
425 state = curr_state;
426 }
427 }
428
429 last_state = curr_state;
430}
