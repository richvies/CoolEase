381void cusb_init(void) {
382 if (usb_state == USB_OFF) {
383 // Initialize clocks
384 cusb_clock_init();
385
386 // Reset USB peripheral
387 SET_REG(USB_CNTR_REG, USB_CNTR_FRES);
388 SET_REG(USB_CNTR_REG, 0);
389 SET_REG(USB_ISTR_REG, 0);
390
391 // Initialize USB Hardware and activate pullup on DP line
392 usbd_dev =
393 usbd_init(&st_usbfs_v2_usb_driver, &dev_desc, &cfg_desc,
394 usb_strings, sizeof(usb_strings) / sizeof(const char*),
395 usbd_control_buffer, sizeof(usbd_control_buffer));
396
397 // Register Reset Callback
398 usbd_register_reset_callback(usbd_dev, cusb_reset_callback);
399
400 // Register Configuration Callback for HID
401 usbd_register_set_config_callback(usbd_dev, hid_set_config);
402
403 usb_state = USB_INIT;
404
405 // Enable NVIC interrupt (through EXTI18 which is enabled on reset)
406 nvic_enable_irq(NVIC_USB_IRQ);
407 nvic_set_priority(NVIC_USB_IRQ, IRQ_PRIORITY_USB);
408
409 usb_plugged_in = false;
410 }

450 // Enable the VREF for HSI48
451 rcc_periph_clock_enable(RCC_SYSCFG);
452 SYSCFG_CFGR3 |= 0x01;
453 while (!(SYSCFG_CFGR3 & SYSCFG_CFGR3_VREFINT_RDYF)) {

457 }

535static void hid_out_report_callback(usbd_device* dev, uint8_t ea) {
536 // serial_printf("o\n");
537
538 // Bit field indicating which pages have been succesfully written
539 static uint32_t page_written[16];
540 static uint32_t page_num;
541 static uint32_t crc_lower;
542 static uint32_t crc_upper;
543 static bool lower;
544 static bool prog_error;
545
546 static uint32_t crc_test_len;
547
548 usbd_ep_read_packet(dev, ea, hid_out_report.buf, HID_REPORT_SIZE_BYTES);
549
550 // Report is a half page of data
551 if (usb_state == USB_PROG_PAGE) {
552 if (!prog_error) {
553 if (boot_program_half_page(lower, lower ? crc_lower : crc_upper,
554 page_num, hid_out_report.buf)) {
555 page_written[page_num / 32] |= (1 << (page_num % 32));
556 }
557 }
558
559 lower = !lower;
560
561 // 2 half pages done
562 if (lower) {
563 usb_state = USB_CONNECTED;
564 }
565 } else if (usb_state == USB_TEST_CRC) {
566 serial_printf("page %u\n", crc_test_len);
567 if (crc_test_len) {
568 crc_lower = ~crc_calculate_block(hid_out_report.buf, 16);
569 --crc_test_len;
570 }
571
572 if (crc_test_len == 0) {
573 usb_state = USB_CONNECTED;
574 serial_printf("Test End %8x\n", crc_lower);
575
576 // Deinit
577 crc_reset();
578 rcc_periph_clock_disable(RCC_CRC);
579 }
580 } else {
581 uint32_t command = hid_out_report.command;
582
583 switch (command) {
584 // Reset
585 case CMD_RESET:
586 usb_state = USB_CONNECTED;
587
588 serial_printf("Reset\n");
589 break;
590
591 // Get Log
592 case CMD_GET_LOG:
593 usb_state = USB_GET_LOG;
594 log_read_reset();
595
596 // Have to write a packet back here to begin IN transfer
597 uint8_t buf[HID_REPORT_SIZE_BYTES] = "Start IN\n";
598 usbd_ep_write_packet(dev, ea, buf, HID_REPORT_SIZE_BYTES);
599
600 serial_printf("Get Log\n");
601 break;
602
603 // Setup for programming
604 case CMD_PROG_START:
605 page_num = 0;
606 for (uint8_t i = 0; i < 16; i++) {
607 // Default all pages to fail, any lost in transit automatically
608 // counted
609 page_written[i] = 0;
610 }
611
612 serial_printf("Prog Start\n");
613 break;
614
615 // Two half pages will be sent next, lower half page first
616 case CMD_PROG_PAGE:
617 usb_state = USB_PROG_PAGE;
618 crc_lower = hid_out_report.crc_lower;
619 crc_upper = hid_out_report.crc_upper;
620 lower = true;
621 prog_error = false;
622
623 page_num = hid_out_report.page_num;
624
625 // Erase page first
626 if (!mem_flash_erase_page(FLASH_APP_ADDRESS +
627 (page_num * FLASH_PAGE_SIZE))) {
628 log_error(ERR_USB_PAGE_ERASE_FAIL);
629 prog_error = true;
630 }
631
632 serial_printf(".page %u\n", page_num);
633 break;
634
635 // Programming end, send back succesful pages
636 // Fails will be resent
637 case CMD_PROG_END:
638 usb_state = USB_CONNECTED;
639
640 usbd_ep_write_packet(dev, ea, page_written, HID_REPORT_SIZE_BYTES);
641
642 serial_printf("Prog End\n");
643 break;
644
645 case CMD_PRINT_START:
646 usb_state = USB_PRINT;
647
648 serial_printf("Print Start\n");
649 break;
650
651 case CMD_PRINT_END:
652 usb_state = USB_CONNECTED;
653
654 serial_printf("Print End\n");
655 break;
656
657 case CMD_TEST_CRC:
658 serial_printf("Test CRC\n%u half pages\ncrc %8x\n",
659 hid_out_report.page_num, hid_out_report.crc_lower);
660
661 usb_state = USB_TEST_CRC;
662
663 crc_test_len = hid_out_report.page_num;
664
665 // Initialize CRC Peripheral
666 rcc_periph_clock_enable(RCC_CRC);
667 crc_reset();
668 crc_set_reverse_input(CRC_CR_REV_IN_BYTE);
669 crc_reverse_output_enable();
670 CRC_INIT = 0xFFFFFFFF;
671
672 break;
673
674 case CMD_END:
675 cusb_end();
676
677 serial_printf("End\n");
678 break;
679
680 default:
681 break;
682 }
683 }
684
685 uint32_t command = hid_out_report.command;
686 serial_printf("command: %d", command);
687}

704void usb_isr(void) {
705 // This print is definitley not the problem with hidapi write failing
706 // serial_printf("I");
707
708 uint16_t istr = *USB_ISTR_REG;
709
710 if (istr & USB_ISTR_RESET) {
711 USB_CLR_ISTR_RESET();
712 usbd_dev->pm_top = USBD_PM_TOP;
713 _usbd_reset(usbd_dev);
714 return;
715 }
716
717 if (istr & USB_ISTR_CTR) {
718 uint8_t ep = istr & USB_ISTR_EP_ID;
719 uint8_t type;
720
721 if (istr & USB_ISTR_DIR) {
722 /* OUT or SETUP? */
723 if (*USB_EP_REG(ep) & USB_EP_SETUP) {
724 type = USB_TRANSACTION_SETUP;
725 st_usbfs_ep_read_packet(usbd_dev, ep,
726 &usbd_dev->control_state.req, 8);
727 } else {
728 type = USB_TRANSACTION_OUT;
729 }
730 } else {
731 type = USB_TRANSACTION_IN;
732 USB_CLR_EP_TX_CTR(ep);
733 }
734
735 if (usbd_dev->user_callback_ctr[ep][type]) {
736 usbd_dev->user_callback_ctr[ep][type](usbd_dev, ep);
737 } else {
738 USB_CLR_EP_RX_CTR(ep);
739 }
740 }
741
742 if (istr & USB_ISTR_SUSP) {
743 USB_CLR_ISTR_SUSP();
744 if (usbd_dev->user_callback_suspend) {
745 usbd_dev->user_callback_suspend();
746 }
747 }
748
749 if (istr & USB_ISTR_WKUP) {
750 USB_CLR_ISTR_WKUP();
751 if (usbd_dev->user_callback_resume) {
752 usbd_dev->user_callback_resume();
753 }
754 }
755
756 if (istr & USB_ISTR_SOF) {
757 USB_CLR_ISTR_SOF();
758 if (usbd_dev->user_callback_sof) {
759 usbd_dev->user_callback_sof();
760 }
761 }
762
763 if (usbd_dev->user_callback_sof) {
764 *USB_CNTR_REG |= USB_CNTR_SOFM;
765 } else {
766 *USB_CNTR_REG &= ~USB_CNTR_SOFM;
767 }
768}
