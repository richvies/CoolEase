418void cusb_init(void) {
419 if (usb_state == USB_OFF) {
420 // Initialize clocks
421 cusb_clock_init();
422
423 // Reset USB peripheral
424 SET_REG(USB_CNTR_REG, USB_CNTR_FRES);
425 SET_REG(USB_CNTR_REG, 0);
426 SET_REG(USB_ISTR_REG, 0);
427
428 // Initialize USB Hardware and activate pullup on DP line
429 usbd_dev =
430 usbd_init(&st_usbfs_v2_usb_driver, &dev_desc, &cfg_desc,
431 usb_strings, sizeof(usb_strings) / sizeof(const char*),
432 usbd_control_buffer, sizeof(usbd_control_buffer));
433
434 // Register Reset Callback
435 usbd_register_reset_callback(usbd_dev, cusb_reset_callback);
436
437 // Register Configuration Callback for HID
438 usbd_register_set_config_callback(usbd_dev, hid_set_config);
439
440 usb_state = USB_INIT;
441
442 // Enable NVIC interrupt (through EXTI18 which is enabled on reset)
443 nvic_enable_irq(NVIC_USB_IRQ);
444 nvic_set_priority(NVIC_USB_IRQ, IRQ_PRIORITY_USB);
445
446 usb_plugged_in = false;
447 }

499 SYSCFG_CFGR3 |= 0x01;
500 while (!(SYSCFG_CFGR3 & SYSCFG_CFGR3_VREFINT_RDYF)) {

582static void hid_out_report_callback(usbd_device* dev, uint8_t ea) {
583 // serial_printf("o\n");
584
585 // Bit field indicating which pages have been succesfully written
586 static uint32_t page_written[16];
587 static uint32_t page_num;
588 static uint32_t crc_lower;
589 static uint32_t crc_upper;
590 static bool lower;
591 static bool prog_error;
592
593 static uint32_t crc_test_len;
594
595 usbd_ep_read_packet(dev, ea, hid_out_report.buf, HID_REPORT_SIZE_BYTES);
596
597 // Report is a half page of data
598 if (usb_state == USB_PROG_PAGE) {
599 if (!prog_error) {
600 if (boot_program_half_page(lower, lower ? crc_lower : crc_upper,
601 page_num, hid_out_report.buf)) {
602 page_written[page_num / 32] |= (1 << (page_num % 32));
603 }
604 }
605
606 lower = !lower;
607
608 // 2 half pages done
609 if (lower) {
610 usb_state = USB_CONNECTED;
611 }
612 } else if (usb_state == USB_TEST_CRC) {
613 serial_printf("page %u\n", crc_test_len);
614 if (crc_test_len) {
615 crc_lower = ~crc_calculate_block(hid_out_report.buf, 16);
616 --crc_test_len;
617 }
618
619 if (crc_test_len == 0) {
620 usb_state = USB_CONNECTED;
621 serial_printf("Test End %8x\n", crc_lower);
622
623 // Deinit
624 crc_reset();
625 rcc_periph_clock_disable(RCC_CRC);
626 }
627 } else {
628 uint32_t command = hid_out_report.command;
629
630 switch (command) {
631 // Reset
632 case CMD_RESET:
633 usb_state = USB_CONNECTED;
634
635 serial_printf("Reset\n");
636 break;
637
638 // Get Log
639 case CMD_GET_LOG:
640 usb_state = USB_GET_LOG;
641 log_read_reset();
642
643 // Have to write a packet back here to begin IN transfer
644 uint8_t buf[HID_REPORT_SIZE_BYTES] = "Start IN\n";
645 usbd_ep_write_packet(dev, ea, buf, HID_REPORT_SIZE_BYTES);
646
647 serial_printf("Get Log\n");
648 break;
649
650 // Setup for programming
651 case CMD_PROG_START:
652 page_num = 0;
653 for (uint8_t i = 0; i < 16; i++) {
654 // Default all pages to fail, any lost in transit automatically
655 // counted
656 page_written[i] = 0;
657 }
658
659 serial_printf("Prog Start\n");
660 break;
661
662 // Two half pages will be sent next, lower half page first
663 case CMD_PROG_PAGE:
664 usb_state = USB_PROG_PAGE;
665 crc_lower = hid_out_report.crc_lower;
666 crc_upper = hid_out_report.crc_upper;
667 lower = true;
668 prog_error = false;
669
670 page_num = hid_out_report.page_num;
671
672 // Erase page first
673 if (!mem_flash_erase_page(FLASH_APP_ADDRESS +
674 (page_num * FLASH_PAGE_SIZE))) {
675 log_error(ERR_USB_PAGE_ERASE_FAIL);
676 prog_error = true;
677 }
678
679 serial_printf(".page %u\n", page_num);
680 break;
681
682 // Programming end, send back succesful pages
683 // Fails will be resent
684 case CMD_PROG_END:
685 usb_state = USB_CONNECTED;
686
687 usbd_ep_write_packet(dev, ea, page_written, HID_REPORT_SIZE_BYTES);
688
689 serial_printf("Prog End\n");
690 break;
691
692 case CMD_PRINT_START:
693 usb_state = USB_PRINT;
694
695 serial_printf("Print Start\n");
696 break;
697
698 case CMD_PRINT_END:
699 usb_state = USB_CONNECTED;
700
701 serial_printf("Print End\n");
702 break;
703
704 case CMD_TEST_CRC:
705 serial_printf("Test CRC\n%u half pages\ncrc %8x\n",
706 hid_out_report.page_num, hid_out_report.crc_lower);
707
708 usb_state = USB_TEST_CRC;
709
710 crc_test_len = hid_out_report.page_num;
711
712 // Initialize CRC Peripheral
713 rcc_periph_clock_enable(RCC_CRC);
714 crc_reset();
715 crc_set_reverse_input(CRC_CR_REV_IN_BYTE);
716 crc_reverse_output_enable();
717 CRC_INIT = 0xFFFFFFFF;
718
719 break;
720
721 case CMD_END:
722 cusb_end();
723
724 serial_printf("End\n");
725 break;
726
727 default:
728 break;
729 }
730 }
731
732 uint32_t command = hid_out_report.command;
733 serial_printf("command: %d", command);
734}

751void usb_isr(void) {
752 // This print is definitley not the problem with hidapi write failing
753 // serial_printf("I");
754
755 uint16_t istr = *USB_ISTR_REG;
756
757 if (istr & USB_ISTR_RESET) {
758 USB_CLR_ISTR_RESET();
759 usbd_dev->pm_top = USBD_PM_TOP;
760 _usbd_reset(usbd_dev);
761 return;
762 }
763
764 if (istr & USB_ISTR_CTR) {
765 uint8_t ep = istr & USB_ISTR_EP_ID;
766 uint8_t type;
767
768 if (istr & USB_ISTR_DIR) {
769 /* OUT or SETUP? */
770 if (*USB_EP_REG(ep) & USB_EP_SETUP) {
771 type = USB_TRANSACTION_SETUP;
772 st_usbfs_ep_read_packet(usbd_dev, ep,
773 &usbd_dev->control_state.req, 8);
774 } else {
775 type = USB_TRANSACTION_OUT;
776 }
777 } else {
778 type = USB_TRANSACTION_IN;
779 USB_CLR_EP_TX_CTR(ep);
780 }
781
782 if (usbd_dev->user_callback_ctr[ep][type]) {
783 usbd_dev->user_callback_ctr[ep][type](usbd_dev, ep);
784 } else {
785 USB_CLR_EP_RX_CTR(ep);
786 }
787 }
788
789 if (istr & USB_ISTR_SUSP) {
790 USB_CLR_ISTR_SUSP();
791 if (usbd_dev->user_callback_suspend) {
792 usbd_dev->user_callback_suspend();
793 }
794 }
795
796 if (istr & USB_ISTR_WKUP) {
797 USB_CLR_ISTR_WKUP();
798 if (usbd_dev->user_callback_resume) {
799 usbd_dev->user_callback_resume();
800 }
801 }
802
803 if (istr & USB_ISTR_SOF) {
804 USB_CLR_ISTR_SOF();
805 if (usbd_dev->user_callback_sof) {
806 usbd_dev->user_callback_sof();
807 }
808 }
809
810 if (usbd_dev->user_callback_sof) {
811 *USB_CNTR_REG |= USB_CNTR_SOFM;
812 } else {
813 *USB_CNTR_REG &= ~USB_CNTR_SOFM;
814 }
815}
