254static void sensor(void) {
255 // Initial packet
256 send_packet();
257 report_pend = false;
258 report_wait = 569 + (xorshift32() & 0x3F);
259 serial_printf("%us\n", report_wait);
260
261 timers_set_wakeup_time(SENSOR_SLEEP_TIME);
262 timers_enable_wut_interrupt();
263
264 deinit();
265 set_gpio_for_standby();
266
267 for (;;) {
268 timers_enter_standby();
269
270 // Wakeup
271 if (report_pend) {
272 init();
273 send_packet();
274 report_pend = false;
275 report_wait = 569 + (xorshift32() & 0x3F);
276 serial_printf("%us\n", report_wait);
277
278 deinit();
279 set_gpio_for_standby();
280 }

328static void send_packet(void) {
329 log_printf("Send Packet\n");
330
331 int8_t rf_power = 20;
332
333 /*////////////////////////*/
334 // Update Battery
335 /*////////////////////////*/
336 batt_update_voltages();
337 log_printf("Batt %u\n", batt_get_batt_voltage());
338 batt_end();
339
340 /*////////////////////////*/
341 // Get Average Temperature
342 /*////////////////////////*/
343 uint8_t max_readings = 4;
344 int16_t readings[4] = {22222, 22222, 22222, 22222};
345 tmp112_init();
346 tmp112_read_temperature(readings, max_readings);
347 tmp112_end();
348
349 int32_t sum = 0;
350 uint8_t num_readings = 0;
351 int16_t temp_avg = 22222;
352 for (int i = 0; i < max_readings; i++) {
353 if (readings[i] != 22222) {
354 sum += readings[i];
355 num_readings++;
356 }
357 }
358 // Prevent divide by zero
359 if (num_readings) {
360 temp_avg = sum / num_readings;
361 }
362 log_printf("Temp: %i\n", temp_avg);
363
364 /*////////////////////////*/
365 // Assemble & Encrypt Packet
366 /*////////////////////////*/
367 rfm_packet_t packet;
368 packet.data.device_number = app_info->dev_id;
369 packet.data.battery = batt_get_batt_voltage();
370 packet.data.temperature = temp_avg;
371 packet.data.msg_number = 0;
372 packet.data.bad_reboot = bad_reboot;
373 packet.data.power = rf_power;
374
375 aes_ecb_encrypt(packet.data.buffer);
376
377 /*////////////////////////*/
378 // Send Packet
379 /*////////////////////////*/
380 rfm_init();
381 rfm_config_for_lora(RFM_BW_125KHZ, RFM_CODING_RATE_4_5,
382 RFM_SPREADING_FACTOR_128CPS, true, rf_power);
383 rfm_transmit_packet(packet);
384 rfm_end();
385 log_printf("Sent\n");
386}

389static uint32_t xorshift32(void) {
390 static uint32_t state = 0;
391
392 if (state == 0) {
393 state = app_info->dev_id;
394 }
395
396 /* Algorithm "xor" from p. 4 of Marsaglia, "Xorshift RNGs" */
397 state ^= state << 13;
398 state ^= state >> 17;
399 state ^= state << 5;
400
401 return state;
402}

405void rtc_isr(void) {
406 // scb_reset_system();
407
408 exti_reset_request(EXTI20);
409
410 if (RTC_ISR & RTC_ISR_WUTF) {
411 timers_clear_wakeup_flag();

413 }
414
415 timers_pet_dogs();
416 report_timer += SENSOR_SLEEP_TIME;
417
