436 static void cusb_clock_init(void)
437 {
438  // Set flash, 16Mhz -> 0 waitstates
439  flash_set_ws(FLASH_ACR_LATENCY_0WS);
440 
441  // Turn on HSI16 osc
442  rcc_osc_on(RCC_HSI16);
443  rcc_wait_for_osc_ready(RCC_HSI16);
444 
445  // Select CPU Clock
446  rcc_set_sysclk_source(RCC_HSI16);
447 
448  // HSI is now the wakeup clock
449  RCC_CFGR |= RCC_CFGR_STOPWUCK_HSI16;
450 
451  // Set prescalers for AHB, APB1, APB2
452  rcc_set_hpre(RCC_CFGR_HPRE_NODIV); // AHB -> 16Mhz
453  rcc_set_ppre1(RCC_CFGR_PPRE1_NODIV); // APB1 ->16Mhz
454  rcc_set_ppre2(RCC_CFGR_PPRE2_NODIV); // APB2 ->16Mhz
455 
456  // Set Peripheral Clock Frequencies used
457  rcc_ahb_frequency = 16000000;
458  rcc_apb1_frequency = 16000000;
459  rcc_apb2_frequency = 16000000;
460 
461  // Enable the VREF for HSI48
462  rcc_periph_clock_enable(RCC_SYSCFG);
463  SYSCFG_CFGR3 |= 0x01;
464  while (!(SYSCFG_CFGR3 & SYSCFG_CFGR3_VREFINT_RDYF))
465  {
466  }
467  SYSCFG_CFGR3 |= SYSCFG_CFGR3_ENREF_HSI48;
468  while (!(SYSCFG_CFGR3 & SYSCFG_CFGR3_REF_HSI48_RDYF))
469  {
470  }
471 
472  // Enable HSI48
473  rcc_osc_on(RCC_HSI48);
474  rcc_wait_for_osc_ready(RCC_HSI48);
475 
476  rcc_periph_clock_enable(RCC_USB);
477  rcc_periph_clock_enable(RCC_CRS);
478 
479  // Select RC HSI48 as usb clock
480  rcc_set_hsi48_source_rc48();
481 
482  // turn off MSI
483  rcc_osc_off(RCC_MSI);
484 }

580 void usb_isr(void)
581 {
582  // This print is definitley not the problem with hidapi write failing
583  serial_printf("I");
584 
585  uint16_t istr = *USB_ISTR_REG;
586 
587  if (istr & USB_ISTR_RESET)
588  {
589  USB_CLR_ISTR_RESET();
590  usbd_dev->pm_top = USBD_PM_TOP;
591  _usbd_reset(usbd_dev);
592  return;
593  }
594 
595  if (istr & USB_ISTR_CTR)
596  {
597  uint8_t ep = istr & USB_ISTR_EP_ID;
598  uint8_t type;
599 
600  if (istr & USB_ISTR_DIR)
601  {
602  /* OUT or SETUP? */
603  if (*USB_EP_REG(ep) & USB_EP_SETUP)
604  {
605  type = USB_TRANSACTION_SETUP;
606  st_usbfs_ep_read_packet(usbd_dev, ep, &usbd_dev->control_state.req, 8);
607  }
608  else
609  {
610  type = USB_TRANSACTION_OUT;
611  }
612  }
613  else
614  {
615  type = USB_TRANSACTION_IN;
616  USB_CLR_EP_TX_CTR(ep);
617  }
618 
619  if (usbd_dev->user_callback_ctr[ep][type])
620  {
621  usbd_dev->user_callback_ctr[ep][type](usbd_dev, ep);
622  }
623  else
624  {
625  USB_CLR_EP_RX_CTR(ep);
626  }
627  }
628 
629  if (istr & USB_ISTR_SUSP)
630  {
631  USB_CLR_ISTR_SUSP();
632  if (usbd_dev->user_callback_suspend)
633  {
634  usbd_dev->user_callback_suspend();
635  }
636  }
637 
638  if (istr & USB_ISTR_WKUP)
639  {
640  USB_CLR_ISTR_WKUP();
641  if (usbd_dev->user_callback_resume)
642  {
643  usbd_dev->user_callback_resume();
644  }
645  }
646 
647  if (istr & USB_ISTR_SOF)
648  {
649  USB_CLR_ISTR_SOF();
650  if (usbd_dev->user_callback_sof)
651  {
652  usbd_dev->user_callback_sof();
653  }
654  }
655 
656  if (usbd_dev->user_callback_sof)
657  {
658  *USB_CNTR_REG |= USB_CNTR_SOFM;
659  }
660  else
661  {
662  *USB_CNTR_REG &= ~USB_CNTR_SOFM;
663  }
664 }
