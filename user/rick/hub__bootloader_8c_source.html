884 static bool check_crc(uint32_t expected) {
885  serial_printf("Check CRC\n");
886 
887  bool ret = false;
888  uint32_t file_size = sim800.http.response_size - BIN_HEADER_SIZE;
889 
890  uint32_t crc = 0;
891 
892  if (file_size) {
893  // Todo: make sure num half pages is an even integer, otherwise will
894  // program garbage at end
895  uint16_t num_half_pages =
896  ((file_size - 1) / (FLASH_HALF_PAGE_SIZE_BYTES)) + 1;
897 
898  // Initialize CRC Peripheral
899  rcc_periph_clock_enable(RCC_CRC);
900  crc_reset();
901  crc_set_reverse_input(CRC_CR_REV_IN_BYTE);
902  crc_reverse_output_enable();
903  CRC_INIT = 0xFFFFFFFF;
904 
905  // Get data and calc crc
906  for (uint16_t n = 0; n < num_half_pages; n++) {
907  timers_pet_dogs();
908 
909  // Half page buffer
910  // Using union so that data can be read as bytes and programmed as
911  // u32 this automatically deals with endianness
912  union {
913  uint8_t buf8[FLASH_HALF_PAGE_SIZE_BYTES];
914  uint32_t buf32[(FLASH_HALF_PAGE_SIZE_BYTES) / 4];
915  } half_page;
916 
917  // HTTPREAD command & get number of bytes read
918  // *number of bytes returned may be less than requested
919  // depending how many are left in file SIM800 signifies how
920  // many bytes are returned
921  if (sim_http_read_response(
922  BIN_HEADER_SIZE + (n * FLASH_HALF_PAGE_SIZE_BYTES),
923  FLASH_HALF_PAGE_SIZE_BYTES, half_page.buf8) != 64) {
924  log_printf(".sim resp not 64 bytes\n");
925  return false;
926  }
927 
928  // Add half page to crc
929  crc = ~crc_calculate_block(half_page.buf32, 16);
930  }
931 
932  // Deinit
933  crc_reset();
934  rcc_periph_clock_disable(RCC_CRC);
935 
936  ret = (crc == expected) ? true : false;
937  }
938 
939  serial_printf(".crc %8x %8x\n", crc, expected);
940 
941  return ret;
942 }
